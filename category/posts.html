import numpy as np
import matplotlib.pyplot as plt
from typing import Dict, Tuple

DEFAULT_LENGTHSCALE = 0.1
DEFAULT_N_TRAIN = 10
DEFAULT_N_TEST = 50
DEFAULT_NOISE = 0.00005


def kernel(x1: np.ndarray, x2: np.ndarray, lengthscale: float = DEFAULT_LENGTHSCALE) -> np.ndarray:
    """GP squared exponential kernel.

    Parameters
    ----------
    x1 : np.ndarray
        First N-dimensional array used when creating covariance matrix.
    x2 : np.ndarray
        Second N-dimensional array used when creating covariance matrix.
    lengthscale : float
        Characteristic lengthscale of the squared exponential kernel.
        Default is 0.1 (for demonstration purposes).
    
    Returns
    -------
    gram_matrix : np.ndarray
        Covariance matrix given by the squared exponential kernel.
    """
    sqdist = np.sum(x1**2,1).reshape(-1,1) + np.sum(x2**2,1) - 2*np.dot(x1, x2.T)
    gram_matrix = np.exp(-.5 * (1/lengthscale) * sqdist)
    return gram_matrix


def simulate_toy_dataset(
    n_train: int = DEFAULT_N_TRAIN, 
    n_test: int = DEFAULT_N_TEST, 
    noise: float = DEFAULT_NOISE
) -> Tuple[np.ndarray, np.ndarray, np.ndarray]:
    """Function to generate toy dataset to demo GP.
    
    REMINDER: We're 'training' this GP analytically, meaning we'll have to invert
    the Gram matrix of test points against themselves (N_train x N_train) so let's
    keep N_train small for now).

    Parameters
    ----------
    n_train : int
        Number of training points. Default is 10.
    n_test : int = DEFAULT_N_TEST
        Number of test points. Default is 50.
    noise : float
        Amount of residual noise to add to true function values. Default is 0.00005.
    
    Returns
    -------
    X_train, y_train, X_test : Tuple[np.ndarray, np.ndarray, np.ndarray]
        Tuple of simulated data to demo our GP on.
    """
    # Function we're going to approximate with a GP.
    f = lambda x: np.sin(0.9*x).flatten()

    # Sample some input points and noisy versions of the function evaluated at these points. 
    X_train = np.random.uniform(-5, 5, size=(n_train, 1))
    y_train = f(X_train) + noise*np.random.randn(n_train)

    # Points we're going to make predictions at.
    X_test = np.linspace(-5, 5, n_train).reshape(-1, 1)

    return X_train, y_train, X_test

def fit_gp(
    X_train: np.ndarray, 
    y_train: np.ndarray, 
    X_test: np.ndarray, 
    noise: float = DEFAULT_NOISE
) -> Dict[str, np.ndarray]:
    """Uses the squared exponential kernel to calculate the mean and variance at test points.

    Parameters
    ----------
    X_train : np.ndarray
        Simulated training inputs.
    y_train : np.ndarray
        Simulated training outputs.
    X_test : np.ndarray
        Simulated test inputs.
    noise : float
        Default prior variance to use when 'fitting' the GP. 
        In reality we'd want to tune this, but since we're just farting
        around for now we'll use the default from above (0.00005).

    Returns
    -------
    gp_dict : Dict[str, np.ndarray]
        Dictionary containing:
        - mu : test mean.
        - s : test standard deviation.
        - f_prior : samples from GP prior.
        - f_post : samples from GP posterior.
    """
    n_train = X_train.shape[0]
    K = kernel(X_train, X_train)
    L = np.linalg.cholesky(K + noise * np.eye(n_train))

    Lk = np.linalg.solve(L, kernel(X_train, X_test))
    mu = np.dot(Lk.T, np.linalg.solve(L, y_train))
    
    K_ = kernel(X_test, X_test)
    s2 = np.diag(K_) - np.sum(Lk**2, axis=0)
    s = np.sqrt(s2)

    L = np.linalg.cholesky(K_ + 1e-6 * np.eye(n_train) - np.dot(Lk.T, Lk))
    f_post = mu.reshape(-1, 1) + np.dot(L, np.random.normal(size=(n_train, 10)))
    
    # Calculate GP prior for demo purposes
    L = np.linalg.cholesky(K_ + 1e-6 * np.eye(n_train))
    f_prior = np.dot(L, np.random.normal(size=(n_train, 10)))

    gp_dict = {"mu": mu, "s": s, "f_prior": f_prior, "f_post": f_post}
    return gp_dict

f = lambda x: np.sin(0.9*x).flatten()
X_train, y_train, X_test = simulate_toy_dataset()
gp_dict = fit_gp(X_train=X_train, y_train=y_train, X_test=X_test)

# Plot mean and standard deviation.
plt.figure(1)
plt.clf()
plt.plot(X_train, y_train, 'r+', ms=20)
plt.plot(X_test, f(X_test), 'b-')
plt.gca().fill_between(X_test.flat, gp_dict["mu"] - 2 * gp_dict["s"], gp_dict["mu"] + 2 * gp_dict["s"], color="#dddddd")
plt.plot(X_test, gp_dict["mu"], 'r--', lw=2)
plt.title('Mean Predictions +/- 2$\sigma$')
plt.axis([-5, 5, -3, 3])

# Draw samples from the prior at our test points.
plt.figure(2)
plt.clf()
plt.plot(X_test, gp_dict["f_prior"])
plt.title("Ten Samples from the GP Prior")
plt.axis([-5, 5, -3, 3])
plt.show()

# Draw samples from the posterior at our test points.
plt.figure(3)
plt.clf()
plt.plot(X_test, gp_dict["f_post"])
plt.title("Ten Samples from the GP Posterior")
plt.axis([-5, 5, -3, 3])
plt.show()
